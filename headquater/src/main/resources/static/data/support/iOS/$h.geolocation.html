(static) watchPosition(callback, options) → {String}

Source:
  • wmatrix.js, line 1418

geolocation 플러그인을 통해서 위치가변경될때 마다 callback함수를 통해서 위치에 위도,경도 전달

Example
var watchId = "";
var callback = function(position){
  if (position != null && position.coords != null) {
       var lat = position.coords.latitude;
       var long = position.coords.longitude;

       console.log("watchPositionID:" + watchID + " lat : " + lat + ", long : " + long);
   }
};
var options = { maximumAge: 3000, timeout: 5000, enableHighAccuracy: true };
watchId = $h.geolocation.watchPosition(callback,options);
